                       (&S#KEY *NE '2'))  THEN(DO)

        CALL       PGM(ARE037RA)
        CLOF       OPNID(SARCVFL1)
        DLTOVR     FILE(*ALL)
        ENDDO

          CALL    PGM(ARE037RB)
        ENDDO
