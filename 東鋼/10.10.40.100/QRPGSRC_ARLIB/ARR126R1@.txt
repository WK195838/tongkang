     C*****************************************************************
     C*          KET  LIST
     C*****************************************************************
     C*FILE => SARVIN(營業繳款發票沖銷檔)
     C           K#SRIN    KLIST
     C                     KFLD           SIRVNO
     C                     KFLD           SIITEM
     C*****************************************************************
     C*         主程式開始
     C*****************************************************************
     C  N99                EXSR INISR
     C   L1                EXSR L1CLR
     C                     EXSR DTLSR
     CL1 99                EXSR L1SR
     CLR                   RETRN
     C*****************************************************************
     C           INISR     BEGSR
     C*****************************************************************
     C                     MOVE *ON       *IN99
     C*
     C                     ENDSR
     C*****************************************************************
     C           L1CLR     BEGSR
     C*****************************************************************
     C                     CLEARRAR126F
     C                     MOVEL*BLANK    W#FLAG  1
     C                     MOVEL*BLANK    W#NTCN  1        未兌現碼
     C                     MOVEL*BLANK    W#NTCY  1        已兌現碼
     C*
     C                     ENDSR
     C*****************************************************************
     C           DTLSR     BEGSR
     C*****************************************************************
     C           RVNO      IFEQ 'P60017'
     C                     MOVEL'X'       W#XXXX  1
     C                     ENDIF
     C*
     C*先判斷此張繳款書是否有沖應收帳款
     C*
     C           RVNO      CHAINSIREC                40
     C           *IN40     IFEQ *OFF
     C                     MOVEL'X'       W#FLAG
     C*
     C*一張繳款書內票據若皆未兌現，則排除此張繳款書
     C*若部分兌現則寫入檔案
     C           RVACNO    IFEQ '1131'
     C           RVNTTP    ANDEQ'A'
     C           RVNTCD    IFEQ '1'
     C           RVNTDT    ANDLEU#DATE
     C                     MOVEL'Y'       W#NTCY
     C                     ELSE
     C                     MOVEL'N'       W#NTCN
     C                     ENDIF
     C                     ENDIF
     C*
     C                     ENDIF
     C*
     C                     ENDSR
     C*****************************************************************
     C           L1SR      BEGSR
     C*****************************************************************
     C           W#FLAG    IFNE ' '
     C*
     C           W#NTCY    IFEQ ' '
     C           W#NTCN    ANDNE' '
     C                     GOTO ENDL1
     C                     ENDIF
     C*
     C*先CHECK所沖銷的應收發票日期
     C                     EXSR SR1000
     C*
     C           W#YEAY    IFEQ 0
     C           W#YEAN    ANDNEU#YEAR
     C                     GOTO ENDL1
     C                     ENDIF
     C*
     C                     EXSR SR2000
     C*
     C                     ENDIF
     C*
     C           ENDL1     ENDSR
     C*****************************************************************
     C           SR1000    BEGSR
     C*****************************************************************
     C                     Z-ADD0         W#YEAY  40       符合年度
     C                     Z-ADD0         W#YEAN  40       不符合年度
     C*
     C                     MOVELRVNO      SIRVNO
     C                     Z-ADD0         SIITEM
     C           K#SRIN    SETLLSIREC
     C                     MOVE *OFF      *IN42
     C           *IN42     DOWEQ*OFF
     C                     READ SIREC                    42
     C      42             LEAVE
     C*
     C           SIRVNO    IFNE RVNO
     C                     LEAVE
     C                     ENDIF
     C*
     C           SIINNO    CHAININREC                43
     C  N43      ININDT    IFGE U#YMDS
     C           ININDT    ANDLEU#YMDE
     C                     Z-ADDU#YEAR    W#YEAY
     C                     ELSE
     C                     Z-ADD9999      W#YEAN
     C                     ENDIF
     C*
     C                     ENDDO
     C*
     C                     ENDSR
     C*****************************************************************
     C           SR2000    BEGSR
     C*****************************************************************
     C*寫入暫存檔
     C                     MOVELRVNO      A1RVNO
     C                     Z-ADDRVITEM    A1ITEM
     C                     Z-ADDRVDATE    A1RVDT
     C                     MOVELRVUSTP    A1USTP
     C                     MOVELRVNTTP    A1NTTP
     C                     MOVELRVACTP    A1ACTP
     C                     MOVELRVACNO    A1ACNO
     C                     MOVELRVCUNO    A1CUNO
     C                     MOVELRVCUNM    A1CUNM
     C                     MOVELRVVNO1    A1VNO1
     C                     MOVELRVNTNO    A1NTNO
     C                     Z-ADDRVNTDT    A1NTDT
     C                     Z-ADDRVDUDT    A1DUDT
     C                     Z-ADDRVRAMT    A1AMT1
     C                     WRITERAR126F
     C*
     C*
     C                     ENDSR
